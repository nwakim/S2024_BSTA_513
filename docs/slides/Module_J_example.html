The MLR model

G1 = as.numeric(group==1)
G2 = as.numeric(group==2)
G3 = as.numeric(group==3)
wt = weight - mean(weight)
m1 = lm(hdl~-1+I(G1)+I(G2)+I(G3)+I(G1*wt)+I(G2*wt)+I(G3*wt))
# alternatively: lm(hdl~-1+factor(group)*wt-wt)
summary(m1)$coef
